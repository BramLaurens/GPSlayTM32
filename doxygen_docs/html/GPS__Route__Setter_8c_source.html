39double convert_decimal_degrees(char *nmea_coordinate, char* ns)
40{
41 double raw = atof(nmea_coordinate); // Convert string to double
42
43 int degrees = (int)(raw / 100); // Get the degrees part
44 double minutes = raw - (degrees * 100); // Get the minutes part
45
46 double decimal_degrees = degrees + (minutes / 60.0); // Convert to decimal degrees
47
48 if (ns[0] == 'S' || ns[0] == 'W') // Check if the coordinate is South or West

187uint8_t Remove_Last_Node()
188{
189
190 if(pt_Route == NULL)
191 { // if the linked list is empty then return -1
192 UART_puts("Linked list is empty \r\n");
193 return 2; // error code maybe other nr
194 }
195
196 if(pt_Route->Next_point == NULL)
197 { // the last node of the linked list
198 free(pt_Route); // this frees the memory for the kernel to be used again
199 pt_Route = NULL; // the pointer will still point to the spot in memory so make it NULL so no exidental read write operation is done
200 UART_puts("Linked list is now cleared completly");
201 return 1; // returns 1 when the linked list is completly empty
202 }
203
204 GPS_Route *temp = pt_Route;
205 GPS_Route *prev = NULL;
206
207 while(temp->Next_point != NULL) {
208 prev = temp;
209 temp = temp->Next_point;
210 }
211
212 // temp is now the last node, prev is the second-to-last
213
214 free(temp);
215 prev->Next_point = NULL;
216
217 UART_puts("Removed last node");
218 return 0;
219}

226void Route_Setter(void *argument)
227{
228 //Create a key variable to store the key pressed by the user
229 uint32_t key=0;
230
231 while(TRUE){
232 // Wait for a notification from the ARM key handler.
233 xTaskNotifyWait(0x00, // Don't clear any notification bits on entry
234 0xffffffff, // Clear the notification value on exit
235 &key, // Notified value
236 portMAX_DELAY); // Block indefinitely
237 UART_puts("\r\n");
238
239 switch(key){
240 case 0x01: // key 1 pressed (upper left)
241 #ifdef debug_routesetter
242 UART_puts("\r\n");
243 UART_puts("Trying to set a waypoint...");
244 UART_puts("\r\n");
245 #endif
246 getlatest_GNRMC(&GNRMC_localcopy3);
247 GPS_Route_Maker(); // Creates a node for a linked list everytime the ARM key is pressed
248 break;
249
250 case 0x02: // remove last node
251 UART_puts("Removing last node...");
252 Remove_Last_Node();
253 break;
254
255 case 0x03: // remove all nodes
256 UART_puts("Removing all nodes...");
257 while(Remove_Last_Node() != 0);
258 break;
259
260 case 0x04:
261 UART_puts("Viewing all items in linked list...");
262 View_Linkedlist();
263 break;
264
265 default: // incease a unassigned key is pressed
266 UART_puts("Current key: "); UART_putint(key);
267 UART_puts(" is not in use by GPS_Route_Setter.c");
268 UART_puts("\r\n");
269 break;
270 }
271 osDelay(1);
272 }
273}
