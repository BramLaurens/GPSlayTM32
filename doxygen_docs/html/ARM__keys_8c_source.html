63void ARM_keys_task (void *argument)
64{
65 uint32_t key;

67
68 while(TRUE)
69 {
70 // WAITING FOR users key
71 xTaskNotifyWait (0x00, // Don't clear any notification bits on entry.
72 0xffffffff, // ULONG_MAX, reset the notification value to 0 on exit.
73 &key, // Notified value.
74 portMAX_DELAY); // Block indefinitely.
75
76 xSemaphoreTake(hLED_Sem, portMAX_DELAY); // krijg toegang (mutex) tot leds

81
82 UART_puts("\r\n\tARM_key pressed to leds: "); UART_putint(key);
83
84 if (Uart_debug_out & ARMKEYS_DEBUG_OUT)
85 {
86 UART_puts("\r\n\tARM_key pressed to leds: "); UART_putint(key);
87 }
88
89 xSemaphoreGive(hLED_Sem); // geef toegang (mutex) vrij
90
91 // tot slot, laat de gekleurde ledjes meedoen
92 // maar niet blauw, want die is ingezet voor de timer
93 // kijk naar de manier waarop de if-elses er uitzien
94 for (i=0; i<3; i++)
95 {
96 led = (i==0 ? LEDRED : (i==1 ? LEDORANGE : LEDGREEN));
97 toggle_led(led);
98 }
99
100 // Also notify the Route_Setter task so it can act on this key press.
101 // Using a direct task notification avoids races that happen when multiple

103 osThreadId_t hRoute = GetTaskhandle("Route_setter");
104 if (hRoute)
