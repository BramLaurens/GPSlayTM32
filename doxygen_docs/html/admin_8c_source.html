277void CreateTasks(void)
278{
279 PTASKDATA ptd = tasks; // tasks == &tasks[0]: both are addresses of first struct
280 vTaskSuspendAll(); // zet de scheduler stil en dwing FreeRTOS om eerst alle taken te starten
281
282 for (; ptd->func != NULL; ptd++) // loopje zonder teller; gewoon de pointer steeds ophogen...
283 {
284 UART_puts("\r\nCreated: ");
285 UART_puts((char *)ptd->attr.name);
286
287 // nb. kijk naar de haakjes van argument 3: het gaat om het adres van (structure attr).
288 ptd->hTask = osThreadNew(ptd->func, ptd->argument, &(ptd->attr));
289 }
290
291 UART_puts("\r\n");
292 xTaskResumeAll(); // start nu de scheduler: play ball
293 DisplayTaskData(); // display alle taskdata op UART
294}

353void StartStopTask(int task_id)
354{
355 PTASKDATA ptd = tasks;
356 unsigned int task_nr;
357 TaskStatus_t xTaskDetails;
358
359 for (task_nr = 1; ptd->func != NULL; ptd++, task_nr++)
360 if (task_nr == task_id) // hebbes
361 {
362 vTaskGetInfo (ptd->hTask, &xTaskDetails, pdTRUE, eInvalid);
363 UART_puts("\r\n");
364 UART_puts(ptd->attr.name);
365
366 // een task kan meerdere statussen hebben; het wordt simpeler
367 // als je alleen de eSuspended checkt.
368 if (xTaskDetails.eCurrentState == eSuspended)
369 {
370 vTaskResume(ptd->hTask);
371 UART_puts(" resumed.");
372 }
373 else // bv. eBlocked, eRunning, eInvalid etc.
374 {
375 vTaskSuspend(ptd->hTask);
376 UART_puts(" suspended.");
377 }
378 return;
379 }
380}

387void DisplayTaskData(void)
388{
389 PTASKDATA ptd = tasks;
390 UBaseType_t highwatermark;
391 TaskStatus_t xTaskDetails;
392 unsigned int free, totalalloc = 0;
393
394 unsigned int task_nr; // tasknr for changing priority
395
396 for (task_nr=1; ptd->func != NULL; ptd++, task_nr++)
397 {
398 highwatermark = uxTaskGetStackHighWaterMark(ptd->hTask); // amount of free bytes
399 free = (highwatermark * 100) / ptd->attr.stack_size; // in percentage
400 totalalloc += ptd->attr.stack_size;
401
402 vTaskGetInfo (ptd->hTask, &xTaskDetails, pdTRUE, eInvalid); // get task status
403
404 UART_puts("\r\n\t["); UART_putint(task_nr); UART_puts("] ");
405 UART_puts(ptd->attr.name);
406 UART_puts("\t priority: "); UART_putint(ptd->attr.priority);
407 UART_puts("\t stacksize: "); UART_putint(ptd->attr.stack_size * 4);
408 UART_puts("\t free: "); UART_putint(highwatermark*4);
409 UART_puts("\t used: "); UART_putint(100 - free); UART_puts("%");
410 UART_puts("\t status: "); UART_puts(xTaskDetails.eCurrentState == eSuspended ? "suspended": "running");
411 }
412 UART_puts("\r\n\tTotal heap: "); UART_putint(configTOTAL_HEAP_SIZE);
413 UART_puts(" Allocated task stack: "); UART_putint(totalalloc * 4);
414 UART_puts(" Free heap space: "); UART_putint(configTOTAL_HEAP_SIZE - (totalalloc * 4));
415 UART_puts("\r\n");
416}
